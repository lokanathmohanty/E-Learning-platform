Introduction

JavaScript is a lightweight, interpreted programming language used to create interactive effects within web browsers.

Basic Syntax

// Example
console.log("Hello World!");
let x = 10;

Variables

JavaScript variables can be declared using var, let, or const.

let name = "John";
const age = 25;
var city = "New York";

Functions

function greet(name) {
  return "Hello " + name;
}

const sayHi = (name) => Hi ${name}!;

Arrays

const fruits = ["apple", "banana", "cherry"];
fruits.push("mango");
console.log(fruits[0]);

Objects

const person = {
  name: "John",
  age: 30,
  greet() {
    console.log("Hi, I'm " + this.name);
  }
};
person.greet();

Loops

for (let i = 0; i < 5; i++) {
  console.log(i);
}

let n = 0;
while (n < 3) {
  console.log(n);
  n++;
}

DOM Manipulation

document.getElementById("myId").innerText = "Hello DOM!";
document.querySelector(".myClass").style.color = "blue";

Events

document.querySelector("button").addEventListener("click", () => {
  alert("Button Clicked!");
});

Promises

const fetchData = new Promise((resolve, reject) => {
  setTimeout(() => resolve("Data Loaded"), 2000);
});

fetchData.then(data => console.log(data));
